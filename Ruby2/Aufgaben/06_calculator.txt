class Calculator

	attr_accessor :zahl1, :zahl2

	def initialize(zahl1, zahl2)
		@zahl1 = zahl1
		@zahl2 = zahl2
	end

	def add
		@zahl1 + @zahl2
	end

	def subtract
		@zahl1 - @zahl2
	end

end
